КонтейнерДочернегоКаталога.Строки.Количество() > 0 Тогда				КонтейнерКаталога.Строки.Добавить(КонтейнерДочернегоКаталога);			КонецЕсли;		ИначеЕсли НРег(Файл.Расширение) = ".epf"
